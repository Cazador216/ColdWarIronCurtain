on_actions = {
	on_startup = {
		effect = {
			USA = {
				set_variable = {
					sdiFunds = 0
				}
				clamp_variable = {
					var = sdiProgress
					min = 0
					max = 100
				}
			}
		}
	}
	on_monthly = {
		effect = {
			if = {
				limit = {
					check_variable = { var = sdiProgress value = 99 compare = greater_than }
				}
				USA = {
					set_country_flag = SDI_FINISHED
				}
			}
			if = {
				limit = {
					USA = {
						has_country_flag = SDI_FINISHED
					}
					AND = {
						USA = {
							NOT = {
								has_country_flag = SDI_fc #Finished check
							}
						}
					}
				}
				#Project finished script frame
				set_country_flag = SDI_fc
			}
		}
	}
}
