on_actions = { 
	on_startup = {
		effect = {
			ARG = { 
				set_variable = { var = var_ARG_influence_about_CHL value = 100 }
				set_variable = { var = var_CHL_influence_about_ARG value = 60 }
			}
			ARG = {
				create_dynamic_country = {
					original_tag = ARG
					set_country_flag = CWIC_ARG_civil_war_unification_flag
					set_politics = {
						ruling_party = neutrality ##cambiar
						elections_allowed = no					
					}
					set_popularities = {
						neutrality = 100
					}
					set_research_slots = 3
					reserve_dynamic_country = yes
					
				}
			}

			CHL = {
				create_dynamic_country = {
					original_tag = CHL
					set_politics = {
						ruling_party = neutrality ##cambiar
						elections_allowed = no					
					}
					set_popularities = {
						neutrality = 100
					}
					set_research_slots = 2
					reserve_dynamic_country = yes
					set_country_flag = CWIC_CHL_civil_war_unification_flag
					
				}
			}

			every_state = {
				limit = { is_core_of = ARG }
				remove_core_of = AR1
			}

			every_state = {
				limit = { is_core_of = CHL }
				remove_core_of = CL1
			}

			
		}
	}

	on_capitulation = {
		effect = {
			FROM = {
				if = {
					limit = { 						 
						CL1 = {
							exists = yes
						}
					}
					ARG = {
						set_country_flag = CWIC_CHL_civil_war_unification_flag
						annex_country = {
							target = CL1
							transfer_troops = yes
						}
					}					
					country_event = CWIC_ARG_CHL_union.8
					
				}
				if = {
					limit = { 
						AR1 = {
							exists = yes
						}
					}
					CHL = {
						set_country_flag = CWIC_ARG_civil_war_unification_flag
						annex_country = {
							target = AR1
							transfer_troops = yes
						}
					}					
					country_event = CWIC_ARG_CHL_union.9
					
				}
			}
		}
	}

}
