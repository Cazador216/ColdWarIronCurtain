on_actions = {

	on_startup = {
		effect = {
			NAT = {
				every_country = {
					limit = {
						has_idea = nato_member
					}
					add_to_array = { NAT.NATO_member_countries = THIS.id } 
				}
				set_variable = { NATO_military_cooperation = 45 }
				set_variable = { NATO_political_integration = 0 }
				set_variable = { NATO_espionage_integration = 25 }
				get_sorted_scored_countries = {
					scorer = NATO_ranking_scorer
					array = NATO_member_countries 
				}
				add_to_variable = { NAT.NATO_gui_dirty = 0.1 }
				country_event = {
					id = twrta.1005
				}
			}
		}
	}
	on_monthly_USA = {
		effect = {
			if = {
				limit = {
					NOT = {
						check_variable = { NAT.NATO_political_integration < 20 }
					}
				}
				add_to_variable = { NAT.NATO_political_integration = 0.25 }
				add_to_variable = { NAT.NATO_military_cooperation = 0.25 }
			}
		}
	}
	on_daily_USA = {
		effect = {
			NAT = {
				NATO_recalculate_member_effects = yes

		}
