increase_pensions = {
	if = {
        limit = {
            has_idea = increased_pensions_2
        }   
        add_ideas = increased_pensions_3
        remove_ideas = increased_pensions_2
    }
    else_if = {
        limit = {
            has_idea = increased_pensions
        }   
        add_ideas = increased_pensions_2
        remove_ideas = increased_pensions
    }
    else_if = {
        limit = {
			NOT = {
				has_idea = increased_pensions
			}
        }
        add_ideas = increased_pensions
    }
    else = {
        add_political_power = 100
    }
}

decrease_pensions = {
	if = {
        limit = {
            has_idea = increased_pensions_3
        }   
        add_ideas = increased_pensions_2
        remove_ideas = increased_pensions_3
    }
    else_if = {
        limit = {
            has_idea = increased_pensions_2
        }   
        add_ideas = increased_pensions
        remove_ideas = increased_pensions_2
    }
    else_if = {
		limit = {
			has_idea = increase_pensions
		}
        remove_ideas = increased_pensions
    }
    else = {
        add_political_power = 100
    }
}
