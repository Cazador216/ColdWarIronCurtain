get_corporate_data = {
	set_variable = {
		global.United_Fruit_Company_Control = United_Fruit_Company_Control
	}
	set_variable = {
		global.Standard_Fruit_Company_Control = Standard_Fruit_Company_Control
	}
	set_variable = {
		global.Domestic_Control = Domestic_Control
	}
	set_variable = {
		global.American_Support = American_Support
	}
	set_variable = {
		global.British_Support = British_Support
	}
	if = {
		limit = {
			check_variable = {
				Domestic_Control > United_Fruit_Company_Control
			}
			check_variable = {
				Domestic_Control > Standard_Fruit_Company_Control
			}
		}
		set_variable = {
			LATAM_Corporation_Frame = 3
		}
	}
	else_if = {
		limit = {
			check_variable = {
				Standard_Fruit_Company_Control > United_Fruit_Company_Control
			}
			check_variable = {
				Standard_Fruit_Company_Control > Domestic_Control
			}
		}
		set_variable = {
			LATAM_Corporation_Frame = 2	#SFC
		}
	}
	else = {
		set_variable = {
			LATAM_Corporation_Frame = 1	#UFC
		}
	}
}
