		context_type = player_context

		window_name = "Loan_Confirmation"
		properties = {
			country_flag_for_trade_screen = {
				image = "[this_loan_country.GetFlag]"
			}	
			root_flag_for_trade_screen = {
				image = "[this_loan_country.GetFlag]"
			}	
		}
		visible = {
			check_variable = { show_loan_screen = 1 }
		}	
		triggers = {
			increase_loan_term_click_enabled = {
				check_variable = {
					current_index < max_index
				}
			}
			decrease_loan_term_click_enabled = {
				check_variable = {
					current_index > 0
				}
			}
			increase_loan_principal_click_enabled = {
				check_variable = {
					purchase_amount < CurrentMaxBorrowable
				}
			}
			decrease_loan_principal_click_enabled = {
				check_variable = {
					purchase_amount > 0
				}
			}
		}
		effects = {
			confirm_loan_purchase_click = {
				set_variable = {
					loan_amount = potential_loan_total
				}
				add_to_array = {
					debtTotalsArray = loan_amount
				}
				add_to_array = {
					debtOwnerArray = global_event_target:this_loan_country
				}
				add_to_array = {
					debtOwnerDomesticStatusArray = bank_name_prompt
				}
				add_to_array = {
					debtInterestArray = potential_loan_interest
				}
				set_variable = {
					debtPayment = potential_loan_monthly_cost
				}
				add_to_array = { debtPaymentArray = debtPayment }
				add_to_variable = { currentDebtPayments = debtPayment }
				loan_set_assigning_borrowable = yes
				
				#close screen
				set_variable = {
					show_loan_screen = 0
				}
			}
			cancel_button_click = {
				set_variable = {
					show_loan_screen = 0
				}
			}
			increase_loan_term_click = {
				add_to_variable = { current_index = 1 }
				ROOT = {
					calculate_loan_details = yes
				}
			}
			decrease_loan_term_click = {
				add_to_variable = { current_index = -1 }
				ROOT = {
					calculate_loan_details = yes
				}
			}
			increase_loan_principal_click = {
				add_to_variable = {
					purchase_amount = 1
				}
				ROOT = {
					calculate_loan_details = yes
				}
			}
			decrease_loan_principal_click = {
				add_to_variable = {
					purchase_amount = -1
				}
				ROOT = {
					calculate_loan_details = yes
				}
			}
		}
