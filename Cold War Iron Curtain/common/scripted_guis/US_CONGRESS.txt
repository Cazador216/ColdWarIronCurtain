    Dem_Primary_Negotiation_Window = {
		context_type = player_context
		window_name = "Dem_Primary_Negotiations"   
		visible = {
			check_variable = {
                show_state_negotiatons = 1
            }
		}  
        properties = {
            flag_for_primary_screen_flag = { 
                image = "[GetFlagNegotiationType]"
			}    
            flag_for_primary_screen_2_flag = { 
                image = "[GetFlagNegotiationType]"
			}    
        }    
        triggers = {
            okay_button_click_enabled = {
                check_variable = {
                    negotiation_offers > 0
                }     
                #check_variable = {
                #    Dem_Activity_Points^Playing_As_Display > 0.999
                #}
            }
        } 
        effects = {
            close_button_click = {
                set_variable = {
                    show_state_negotiatons = 0
                }
            }
            okay_button_click = {
                add_to_variable = {
                    Dem_Activity_Points^Playing_As_Display = -1
                }
                if = {
                    limit = {
                        check_variable = {
                            negotiation_offers = 1
                        }
                    }
                    set_variable = {
                        accept_rate = 55
                    }
                    set_variable = {
                        reject_rate = 45
                    }
                }
                else = {
                    set_variable = {
                        accept_rate = 90
                    }
                    set_variable = {
                        reject_rate = 10
                    }
                }
                random_list = {
                    var:accept_rate = {
                        set_variable = {
                            show_dem_neg_response = 1
                        }
                        if = {
                            limit = {
                                check_variable = {
                                    negotiation_toggle^0 = 1
                                }
                                NOT = {
                                    meta_trigger = { 
                                        text = {
                                            is_in_array = { 
                                                dem_campaign_[ARRAY_NUMBER]_promises = dem_state_leader_issues^0 
                                            }
                                        }
                                        ARRAY_NUMBER = "[?Playing_As_Display]"
                                    }    
                                }
                            }
                            meta_effect = { 
                                text = {
                                    add_to_array = { 
                                        dem_campaign_[ARRAY_NUMBER]_promises = dem_state_leader_issues^0
                                    }
                                }
                                ARRAY_NUMBER = "[?Playing_As_Display]"
                            }    
                        }
                        if = {
                            limit = {
                                check_variable = {
                                    negotiation_toggle^1 = 1
                                }
                                NOT = {
                                    meta_trigger = { 
                                        text = {
                                            is_in_array = { dem_campaign_[ARRAY_NUMBER]_promises = dem_state_leader_issues^1 }
                                        }
                                        ARRAY_NUMBER = "[?Playing_As_Display]"
                                    }    
                                }
                            }
                            meta_effect = { 
                                text = {
                                    add_to_array = { 
                                        dem_campaign_[ARRAY_NUMBER]_promises = dem_state_leader_issues^1
                                    }
                                }
                                ARRAY_NUMBER = "[?Playing_As_Display]"
                            }    
                        }
                        set_variable = {
                            show_state_negotiatons = 0
                        }
                        var:USA_DEM_NOM_NEG_TARGET = {
                            if = {
                                limit = {
                                    check_variable = {
                                        Contest_Type = 4
                                    }
                                }
                                set_variable = {    
                                    CurrentDemStateWinner = USA.Playing_As
                                }
                                add_to_variable = {
                                    USA.Democratic_Delegates^USA.Playing_As_Display = Democratic_Delegates
                                }
                                set_variable = {
                                    Dem_Primary_Frame = USA.Playing_As_Position
                                }
                            }
                            else = {
                                set_variable = {
                                    favorite_son_support_transfer = Canidate_Support^ROOT.Favorite_Son_Position
                                }
                                add_to_variable = { 
                                    Canidate_Support^ROOT.Playing_As_Position = favorite_son_support_transfer 
                                }    
                                subtract_from_variable = {
                                    Canidate_Support^ROOT.Favorite_Son_Position = favorite_son_support_transfer
                                }                 
                            }
                            set_variable = {
                                negotiated_dem_primary = 1
                            }
                        }
                        generate_campaign_map_numbers_dem = yes
                    }

                    
                    var:reject_rate = {
                        set_variable = {
                            show_dem_neg_response = 2
                        }
                    }
                }
            }
        } 
    }
